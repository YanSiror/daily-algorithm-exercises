整型数组翻转

    
        /*
         * 输入: 数组 长度
         * 输出: 翻转后的数组
        */
        int * reverseInt(int * nums,int size){
            int i,j;
            for(i = 0;i < size/2;i++){
                int temp = nums[i];
                nums[i] = nums[size-i-1];
                nums[size-i-1] = temp;
            }
            return nums;
        }
